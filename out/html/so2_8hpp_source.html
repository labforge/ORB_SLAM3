209  template <typename OtherDerived>
210  SOPHUS_FUNC SO2Product<OtherDerived> operator*(
211  SO2Base<OtherDerived> const& other) const {
212  using ResultT = ReturnScalar<OtherDerived>;
213  Scalar const lhs_real = unit_complex().x();
214  Scalar const lhs_imag = unit_complex().y();
215  typename OtherDerived::Scalar const& rhs_real = other.unit_complex().x();
216  typename OtherDerived::Scalar const& rhs_imag = other.unit_complex().y();
217  // complex multiplication
218  ResultT const result_real = lhs_real * rhs_real - lhs_imag * rhs_imag;
219  ResultT const result_imag = lhs_real * rhs_imag + lhs_imag * rhs_real;
220 
221  ResultT const squared_norm =
222  result_real * result_real + result_imag * result_imag;
223  // We can assume that the squared-norm is close to 1 since we deal with a
224  // unit complex number. Due to numerical precision issues, there might
225  // be a small drift after pose concatenation. Hence, we need to renormalizes
226  // the complex number here.
227  // Since squared-norm is close to 1, we do not need to calculate the costly
228  // square-root, but can use an approximation around 1 (see
229  // http://stackoverflow.com/a/12934750 for details).
230  if (squared_norm != ResultT(1.0)) {
231  ResultT const scale = ResultT(2.0) / (ResultT(1.0) + squared_norm);
232  return SO2Product<OtherDerived>(result_real * scale, result_imag * scale);
233  }
234  return SO2Product<OtherDerived>(result_real, result_imag);
235  }
