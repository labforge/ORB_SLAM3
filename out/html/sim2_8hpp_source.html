480  SOPHUS_FUNC static Sim2<Scalar> exp(Tangent const& a) {
481  // For the derivation of the exponential map of Sim(N) see
482  // H. Strasdat, "Local Accuracy and Global Consistency for Efficient Visual
483  // SLAM", PhD thesis, 2012.
484  // http:///hauke.strasdat.net/files/strasdat_thesis_2012.pdf (A.5, pp. 186)
485  Vector2<Scalar> const upsilon = a.segment(0, 2);
486  Scalar const theta = a[2];
487  Scalar const sigma = a[3];
488  RxSO2<Scalar> rxso2 = RxSO2<Scalar>::exp(a.template tail<2>());
489  Matrix2<Scalar> const Omega = SO2<Scalar>::hat(theta);
490  Matrix2<Scalar> const W = details::calcW<Scalar, 2>(Omega, theta, sigma);
491  return Sim2<Scalar>(rxso2, W * upsilon);
492  }

557  SOPHUS_FUNC static Tangent lieBracket(Tangent const& a, Tangent const& b) {
558  Vector2<Scalar> const upsilon1 = a.template head<2>();
559  Vector2<Scalar> const upsilon2 = b.template head<2>();
560  Scalar const theta1 = a[2];
561  Scalar const theta2 = b[2];
562  Scalar const sigma1 = a[3];
563  Scalar const sigma2 = b[3];
564 
565  Tangent res;
566  res[0] = -theta1 * upsilon2[1] + theta2 * upsilon1[1] +
567  sigma1 * upsilon2[0] - sigma2 * upsilon1[0];
568  res[1] = theta1 * upsilon2[0] - theta2 * upsilon1[0] +
569  sigma1 * upsilon2[1] - sigma2 * upsilon1[1];
570  res[2] = Scalar(0);
571  res[3] = Scalar(0);
572 
573  return res;
574  }
