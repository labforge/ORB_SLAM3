311 void Sim3Solver::ComputeSim3(Eigen::Matrix3f &P1, Eigen::Matrix3f &P2)
312 {
313  // Custom implementation of:
314  // Horn 1987, Closed-form solution of absolute orientataion using unit quaternions
315 
316  // Step 1: Centroid and relative coordinates
317 
318  Eigen::Matrix3f Pr1; // Relative coordinates to centroid (set 1)
319  Eigen::Matrix3f Pr2; // Relative coordinates to centroid (set 2)
320  Eigen::Vector3f O1; // Centroid of P1
321  Eigen::Vector3f O2; // Centroid of P2
322 
323  ComputeCentroid(P1,Pr1,O1);
324  ComputeCentroid(P2,Pr2,O2);
325 
326  // Step 2: Compute M matrix
327 
328  Eigen::Matrix3f M = Pr2 * Pr1.transpose();
329 
330  // Step 3: Compute N matrix
331  double N11, N12, N13, N14, N22, N23, N24, N33, N34, N44;
332 
333  Eigen::Matrix4f N;
334 
335  N11 = M(0,0)+M(1,1)+M(2,2);
336  N12 = M(1,2)-M(2,1);
337  N13 = M(2,0)-M(0,2);
338  N14 = M(0,1)-M(1,0);
339  N22 = M(0,0)-M(1,1)-M(2,2);
340  N23 = M(0,1)+M(1,0);
341  N24 = M(2,0)+M(0,2);
342  N33 = -M(0,0)+M(1,1)-M(2,2);
343  N34 = M(1,2)+M(2,1);
344  N44 = -M(0,0)-M(1,1)+M(2,2);
345 
346  N << N11, N12, N13, N14,
347  N12, N22, N23, N24,
348  N13, N23, N33, N34,
349  N14, N24, N34, N44;
350 
351 
352  // Step 4: Eigenvector of the highest eigenvalue
353  Eigen::EigenSolver<Eigen::Matrix4f> eigSolver;
354  eigSolver.compute(N);
355 
356  Eigen::Vector4f eval = eigSolver.eigenvalues().real();
357  Eigen::Matrix4f evec = eigSolver.eigenvectors().real(); //evec[0] is the quaternion of the desired rotation
358 
359  int maxIndex; // should be zero
360  eval.maxCoeff(&maxIndex);
361 
362  Eigen::Vector3f vec = evec.block<3,1>(1,maxIndex); //extract imaginary part of the quaternion (sin*axis)
363 
364  // Rotation angle. sin is the norm of the imaginary part, cos is the real part
365  double ang=atan2(vec.norm(),evec(0,maxIndex));
366  if(vec.norm()!=0){
367  vec = 2*ang*vec/vec.norm(); //Angle-axis representation. quaternion angle is the half
368  }
369  mR12i = Sophus::SO3f::exp(vec).matrix();
370 
371  // Step 5: Rotate set 2
372  Eigen::Matrix3f P3 = mR12i*Pr2;
373 
374  // Step 6: Scale
375 
376  if(!mbFixScale)
377  {
378  double cvnom = Converter::toCvMat(Pr1).dot(Converter::toCvMat(P3));
379  double nom = (Pr1.array() * P3.array()).sum();
380  if (abs(nom-cvnom)>1e-3)
381  std::cout << "sim3 solver: " << abs(nom-cvnom) << std::endl << nom << std::endl;
382  Eigen::Array<float,3,3> aux_P3;
383  aux_P3 = P3.array() * P3.array();
384  double den = aux_P3.sum();
385 
386  ms12i = nom/den;
387  }
388  else
389  ms12i = 1.0f;
390 
391  // Step 7: Translation
392  mt12i = O1 - ms12i * mR12i * O2;
393 
394  // Step 8: Transformation
395 
396  // Step 8.1 T12
397  mT12i.setIdentity();
398 
399  Eigen::Matrix3f sR = ms12i*mR12i;
400  mT12i.block<3,3>(0,0) = sR;
401  mT12i.block<3,1>(0,3) = mt12i;
402 
403 
404  // Step 8.2 T21
405  mT21i.setIdentity();
406  Eigen::Matrix3f sRinv = (1.0/ms12i)*mR12i.transpose();
407 
408  // sRinv.copyTo(mT21i.rowRange(0,3).colRange(0,3));
409  mT21i.block<3,3>(0,0) = sRinv;
410 
411  Eigen::Vector3f tinv = -sRinv * mt12i;
412  mT21i.block<3,1>(0,3) = tinv;
413 }
