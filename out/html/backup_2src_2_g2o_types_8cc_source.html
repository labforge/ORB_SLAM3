642 void EdgeInertialGS::linearizeOplus()
643 {
644  const VertexPose* VP1 = static_cast<const VertexPose*>(_vertices[0]);
645  const VertexVelocity* VV1= static_cast<const VertexVelocity*>(_vertices[1]);
646  const VertexGyroBias* VG= static_cast<const VertexGyroBias*>(_vertices[2]);
647  const VertexAccBias* VA= static_cast<const VertexAccBias*>(_vertices[3]);
648  const VertexPose* VP2 = static_cast<const VertexPose*>(_vertices[4]);
649  const VertexVelocity* VV2 = static_cast<const VertexVelocity*>(_vertices[5]);
650  const VertexGDir* VGDir = static_cast<const VertexGDir*>(_vertices[6]);
651  const VertexScale* VS = static_cast<const VertexScale*>(_vertices[7]);
652  const IMU::Bias b(VA->estimate()[0],VA->estimate()[1],VA->estimate()[2],VG->estimate()[0],VG->estimate()[1],VG->estimate()[2]);
653  const IMU::Bias db = mpInt->GetDeltaBias(b);
654 
655  Eigen::Vector3d dbg;
656  dbg << db.bwx, db.bwy, db.bwz;
657 
658  const Eigen::Matrix3d Rwb1 = VP1->estimate().Rwb;
659  const Eigen::Matrix3d Rbw1 = Rwb1.transpose();
660  const Eigen::Matrix3d Rwb2 = VP2->estimate().Rwb;
661  const Eigen::Matrix3d Rwg = VGDir->estimate().Rwg;
662  Eigen::MatrixXd Gm = Eigen::MatrixXd::Zero(3,2);
663  Gm(0,1) = -IMU::GRAVITY_VALUE;
664  Gm(1,0) = IMU::GRAVITY_VALUE;
665  const double s = VS->estimate();
666  const Eigen::MatrixXd dGdTheta = Rwg*Gm;
667  const Eigen::Matrix3d dR = mpInt->GetDeltaRotation(b).cast<double>();
668  const Eigen::Matrix3d eR = dR.transpose()*Rbw1*Rwb2;
669  const Eigen::Vector3d er = LogSO3(eR);
670  const Eigen::Matrix3d invJr = InverseRightJacobianSO3(er);
671 
672  // Jacobians wrt Pose 1
673  _jacobianOplus[0].setZero();
674  // rotation
675  _jacobianOplus[0].block<3,3>(0,0) = -invJr*Rwb2.transpose()*Rwb1;
676  _jacobianOplus[0].block<3,3>(3,0) = Sophus::SO3d::hat(Rbw1*(s*(VV2->estimate() - VV1->estimate()) - g*dt));
677  _jacobianOplus[0].block<3,3>(6,0) = Sophus::SO3d::hat(Rbw1*(s*(VP2->estimate().twb - VP1->estimate().twb
678  - VV1->estimate()*dt) - 0.5*g*dt*dt));
679  // translation
680  _jacobianOplus[0].block<3,3>(6,3) = Eigen::DiagonalMatrix<double,3>(-s,-s,-s);
681 
682  // Jacobians wrt Velocity 1
683  _jacobianOplus[1].setZero();
684  _jacobianOplus[1].block<3,3>(3,0) = -s*Rbw1;
685  _jacobianOplus[1].block<3,3>(6,0) = -s*Rbw1*dt;
686 
687  // Jacobians wrt Gyro bias
688  _jacobianOplus[2].setZero();
689  _jacobianOplus[2].block<3,3>(0,0) = -invJr*eR.transpose()*RightJacobianSO3(JRg*dbg)*JRg;
690  _jacobianOplus[2].block<3,3>(3,0) = -JVg;
691  _jacobianOplus[2].block<3,3>(6,0) = -JPg;
692 
693  // Jacobians wrt Accelerometer bias
694  _jacobianOplus[3].setZero();
695  _jacobianOplus[3].block<3,3>(3,0) = -JVa;
696  _jacobianOplus[3].block<3,3>(6,0) = -JPa;
697 
698  // Jacobians wrt Pose 2
699  _jacobianOplus[4].setZero();
700  // rotation
701  _jacobianOplus[4].block<3,3>(0,0) = invJr;
702  // translation
703  _jacobianOplus[4].block<3,3>(6,3) = s*Rbw1*Rwb2;
704 
705  // Jacobians wrt Velocity 2
706  _jacobianOplus[5].setZero();
707  _jacobianOplus[5].block<3,3>(3,0) = s*Rbw1;
708 
709  // Jacobians wrt Gravity direction
710  _jacobianOplus[6].setZero();
711  _jacobianOplus[6].block<3,2>(3,0) = -Rbw1*dGdTheta*dt;
712  _jacobianOplus[6].block<3,2>(6,0) = -0.5*Rbw1*dGdTheta*dt*dt;
713 
714  // Jacobians wrt scale factor
715  _jacobianOplus[7].setZero();
716  _jacobianOplus[7].block<3,1>(3,0) = Rbw1*(VV2->estimate()-VV1->estimate());
717  _jacobianOplus[7].block<3,1>(6,0) = Rbw1*(VP2->estimate().twb-VP1->estimate().twb-VV1->estimate()*dt);
718 }

785  const double d = sqrt(d2);

800 Eigen::Vector3d LogSO3(const Eigen::Matrix3d &R)
801 {
802  const double tr = R(0,0)+R(1,1)+R(2,2);
803  Eigen::Vector3d w;
804  w << (R(2,1)-R(1,2))/2, (R(0,2)-R(2,0))/2, (R(1,0)-R(0,1))/2;
805  const double costheta = (tr-1.0)*0.5f;
806  if(costheta>1 || costheta<-1)
807  return w;
808  const double theta = acos(costheta);
809  const double s = sin(theta);
810  if(fabs(s)<1e-5)
811  return w;
812  else
813  return theta*w/s;
814 }
