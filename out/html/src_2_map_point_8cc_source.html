329 void MapPoint::ComputeDistinctiveDescriptors()
330 {
331  // Retrieve all observed descriptors
332  vector<cv::Mat> vDescriptors;
333 
334  map<KeyFrame*,tuple<int,int>> observations;
335 
336  {
337  unique_lock<mutex> lock1(mMutexFeatures);
338  if(mbBad)
339  return;
340  observations=mObservations;
341  }
342 
343  if(observations.empty())
344  return;
345 
346  vDescriptors.reserve(observations.size());
347 
348  for(map<KeyFrame*,tuple<int,int>>::iterator mit=observations.begin(), mend=observations.end(); mit!=mend; mit++)
349  {
350  KeyFrame* pKF = mit->first;
351 
352  if(!pKF->isBad()){
353  tuple<int,int> indexes = mit -> second;
354  int leftIndex = get<0>(indexes), rightIndex = get<1>(indexes);
355 
356  if(leftIndex != -1){
357  vDescriptors.push_back(pKF->mDescriptors.row(leftIndex));
358  }
359  if(rightIndex != -1){
360  vDescriptors.push_back(pKF->mDescriptors.row(rightIndex));
361  }
362  }
363  }
364 
365  if(vDescriptors.empty())
366  return;
367 
368  // Compute distances between them
369  const size_t N = vDescriptors.size();
370 
371  float Distances[N][N];
372  for(size_t i=0;i<N;i++)
373  {
374  Distances[i][i]=0;
375  for(size_t j=i+1;j<N;j++)
376  {
377  int distij = ORBmatcher::DescriptorDistance(vDescriptors[i],vDescriptors[j]);
378  Distances[i][j]=distij;
379  Distances[j][i]=distij;
380  }
381  }
382 
383  // Take the descriptor with least median distance to the rest
384  int BestMedian = INT_MAX;
385  int BestIdx = 0;
386  for(size_t i=0;i<N;i++)
387  {
388  vector<int> vDists(Distances[i],Distances[i]+N);
389  sort(vDists.begin(),vDists.end());
390  int median = vDists[0.5*(N-1)];
391 
392  if(median<BestMedian)
393  {
394  BestMedian = median;
395  BestIdx = i;
396  }
397  }
398 
399  {
400  unique_lock<mutex> lock(mMutexFeatures);
401  mDescriptor = vDescriptors[BestIdx].clone();
402  }
403 }
