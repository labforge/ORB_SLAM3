252 void Map::ApplyScaledRotation(const Sophus::SE3f &T, const float s, const bool bScaledVel)
253 {
254  unique_lock<mutex> lock(mMutexMap);
255 
256  // Body position (IMU) of first keyframe is fixed to (0,0,0)
257  Sophus::SE3f Tyw = T;
258  Eigen::Matrix3f Ryw = Tyw.rotationMatrix();
259  Eigen::Vector3f tyw = Tyw.translation();
260 
261  for(set<KeyFrame*>::iterator sit=mspKeyFrames.begin(); sit!=mspKeyFrames.end(); sit++)
262  {
263  KeyFrame* pKF = *sit;
264  Sophus::SE3f Twc = pKF->GetPoseInverse();
265  Twc.translation() *= s;
266  Sophus::SE3f Tyc = Tyw*Twc;
267  Sophus::SE3f Tcy = Tyc.inverse();
268  pKF->SetPose(Tcy);
269  Eigen::Vector3f Vw = pKF->GetVelocity();
270  if(!bScaledVel)
271  pKF->SetVelocity(Ryw*Vw);
272  else
273  pKF->SetVelocity(Ryw*Vw*s);
274 
275  }
276  for(set<MapPoint*>::iterator sit=mspMapPoints.begin(); sit!=mspMapPoints.end(); sit++)
277  {
278  MapPoint* pMP = *sit;
279  pMP->SetWorldPos(s * Ryw * pMP->GetWorldPos() + tyw);
280  pMP->UpdateNormalAndDepth();
281  }
282  mnMapChange++;
283 }

427 void Map::PostLoad(KeyFrameDatabase* pKFDB, ORBVocabulary* pORBVoc/*, map<long unsigned int, KeyFrame*>& mpKeyFrameId*/, map<unsigned int, GeometricCamera*> &mpCams)
428 {

475  mpKFinitial = mpKeyFrameId[mnBackupKFinitialID];
476  }
477

479  {
480  mpKFlowerID = mpKeyFrameId[mnBackupKFlowerID];
481  }
482 
483  mvpKeyFrameOrigins.clear();
484  mvpKeyFrameOrigins.reserve(mvBackupKeyFrameOriginsId.size());
485  for(int i = 0; i < mvBackupKeyFrameOriginsId.size(); ++i)
