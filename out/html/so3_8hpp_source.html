247  SOPHUS_FUNC TangentAndTheta logAndTheta() const {
248  TangentAndTheta J;
249  using std::abs;
250  using std::atan;
251  using std::sqrt;
252  Scalar squared_n = unit_quaternion().vec().squaredNorm();
253  Scalar w = unit_quaternion().w();
254 
255  Scalar two_atan_nbyw_by_n;
256 
263 
264  if (squared_n < Constants<Scalar>::epsilon() * Constants<Scalar>::epsilon()) {
265  // If quaternion is normalized and n=0, then w should be 1;
266  // w=0 should never happen here!
267  SOPHUS_ENSURE(abs(w) >= Constants<Scalar>::epsilon(),
268  "Quaternion (%) should be normalized!",
269  unit_quaternion().coeffs().transpose());
270  Scalar squared_w = w * w;
271  two_atan_nbyw_by_n =
272  Scalar(2) / w - Scalar(2.0/3.0) * (squared_n) / (w * squared_w);
273  J.theta = Scalar(2) * squared_n / w;
274  } else {
275  Scalar n = sqrt(squared_n);
276  if (abs(w) < Constants<Scalar>::epsilon()) {
277  if (w > Scalar(0)) {
278  two_atan_nbyw_by_n = Constants<Scalar>::pi() / n;
279  } else {
280  two_atan_nbyw_by_n = -Constants<Scalar>::pi() / n;
281  }
282  } else {
283  two_atan_nbyw_by_n = Scalar(2) * atan(n / w) / n;
284  }
285  J.theta = two_atan_nbyw_by_n * n;
286  }
287 
288  J.tangent = two_atan_nbyw_by_n * unit_quaternion().vec();
289  return J;
290  }

497  SOPHUS_FUNC static Sophus::Matrix<Scalar, num_parameters, DoF> Dx_exp_x(
498  Tangent const& omega) {
499  using std::cos;
500  using std::exp;
501  using std::sin;
502  using std::sqrt;
503  Scalar const c0 = omega[0] * omega[0];
504  Scalar const c1 = omega[1] * omega[1];
505  Scalar const c2 = omega[2] * omega[2];
506  Scalar const c3 = c0 + c1 + c2;
507 
508  if (c3 < Constants<Scalar>::epsilon()) {
509  return Dx_exp_x_at_0();
510  }
511 
512  Scalar const c4 = sqrt(c3);
513  Scalar const c5 = 1.0 / c4;
514  Scalar const c6 = 0.5 * c4;
515  Scalar const c7 = sin(c6);
516  Scalar const c8 = c5 * c7;
517  Scalar const c9 = pow(c3, -3.0L / 2.0L);
518  Scalar const c10 = c7 * c9;
519  Scalar const c11 = Scalar(1.0) / c3;
520  Scalar const c12 = cos(c6);
521  Scalar const c13 = Scalar(0.5) * c11 * c12;
522  Scalar const c14 = c7 * c9 * omega[0];
523  Scalar const c15 = Scalar(0.5) * c11 * c12 * omega[0];
524  Scalar const c16 = -c14 * omega[1] + c15 * omega[1];
525  Scalar const c17 = -c14 * omega[2] + c15 * omega[2];
526  Scalar const c18 = omega[1] * omega[2];
527  Scalar const c19 = -c10 * c18 + c13 * c18;
528  Scalar const c20 = Scalar(0.5) * c5 * c7;
529  Sophus::Matrix<Scalar, num_parameters, DoF> J;
530  J(0, 0) = -c0 * c10 + c0 * c13 + c8;
531  J(0, 1) = c16;
532  J(0, 2) = c17;
533  J(1, 0) = c16;
534  J(1, 1) = -c1 * c10 + c1 * c13 + c8;
535  J(1, 2) = c19;
536  J(2, 0) = c17;
537  J(2, 1) = c19;
538  J(2, 2) = -c10 * c2 + c13 * c2 + c8;
539  J(3, 0) = -c20 * omega[0];
540  J(3, 1) = -c20 * omega[1];
541  J(3, 2) = -c20 * omega[2];
542  return J;
543  }

583  SOPHUS_FUNC static SO3<Scalar> expAndTheta(Tangent const& omega,
584  Scalar* theta) {
585  SOPHUS_ENSURE(theta != nullptr, "must not be nullptr.");
586  using std::abs;
587  using std::cos;
588  using std::sin;
589  using std::sqrt;
590  Scalar theta_sq = omega.squaredNorm();
591 
592  Scalar imag_factor;
593  Scalar real_factor;
594  if (theta_sq <
595  Constants<Scalar>::epsilon() * Constants<Scalar>::epsilon()) {
596  *theta = Scalar(0);
597  Scalar theta_po4 = theta_sq * theta_sq;
598  imag_factor = Scalar(0.5) - Scalar(1.0 / 48.0) * theta_sq +
599  Scalar(1.0 / 3840.0) * theta_po4;
600  real_factor = Scalar(1) - Scalar(1.0 / 8.0) * theta_sq +
601  Scalar(1.0 / 384.0) * theta_po4;
602  } else {
603  *theta = sqrt(theta_sq);
604  Scalar half_theta = Scalar(0.5) * (*theta);
605  Scalar sin_half_theta = sin(half_theta);
606  imag_factor = sin_half_theta / (*theta);
607  real_factor = cos(half_theta);
608  }
609 
610  SO3 q;
611  q.unit_quaternion_nonconst() =
612  QuaternionMember(real_factor, imag_factor * omega.x(),
613  imag_factor * omega.y(), imag_factor * omega.z());
614  SOPHUS_ENSURE(abs(q.unit_quaternion().squaredNorm() - Scalar(1)) <
615  Sophus::Constants<Scalar>::epsilon(),
616  "SO3::exp failed! omega: %, real: %, img: %",
617  omega.transpose(), real_factor, imag_factor);
618  return q;
619  }
