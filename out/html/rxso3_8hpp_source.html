195  Scalar const vy_sq = quaternion().vec().y() * quaternion().vec().y();
196  Scalar const vz_sq = quaternion().vec().z() * quaternion().vec().z();
197  Scalar const w_sq = quaternion().w() * quaternion().w();

199  Scalar const two_vy = Scalar(2) * quaternion().vec().y();
200  Scalar const two_vz = Scalar(2) * quaternion().vec().z();
201  Scalar const two_vx_vy = two_vx * quaternion().vec().y();
202  Scalar const two_vx_vz = two_vx * quaternion().vec().z();
203  Scalar const two_vx_w = two_vx * quaternion().w();
204  Scalar const two_vy_vz = two_vy * quaternion().vec().z();
205  Scalar const two_vy_w = two_vy * quaternion().w();
206  Scalar const two_vz_w = two_vz * quaternion().w();
207 
208  sR(0, 0) = vx_sq - vy_sq - vz_sq + w_sq;
209  sR(1, 0) = two_vx_vy + two_vz_w;
210  sR(2, 0) = two_vx_vz - two_vy_w;
211

213  sR(1, 1) = -vx_sq + vy_sq - vz_sq + w_sq;
214  sR(2, 1) = two_vx_w + two_vy_vz;
215

217  sR(1, 2) = -two_vx_w + two_vy_vz;
218  sR(2, 2) = -vx_sq - vy_sq + vz_sq + w_sq;
