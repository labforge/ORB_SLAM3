105  SOPHUS_FUNC Adjoint Adj() const {
106  Matrix3<Scalar> const R = rxso3().rotationMatrix();
107  Adjoint res;
108  res.setZero();
109  res.template block<3, 3>(0, 0) = rxso3().matrix();

144  SOPHUS_FUNC Tangent log() const {
145  // The derivation of the closed-form Sim(3) logarithm for is done
146  // analogously to the closed-form solution of the SE(3) logarithm, see
147  // J. Gallier, D. Xu, "Computing exponentials of skew symmetric matrices
148  // and logarithms of orthogonal matrices", IJRA 2002.
149  // https:///pdfs.semanticscholar.org/cfe3/e4b39de63c8cabd89bf3feff7f5449fc981d.pdf
150  // (Sec. 6., pp. 8)
151  Tangent res;
152  auto omega_sigma_and_theta = rxso3().logAndTheta();
153  Vector3<Scalar> const omega =
154  omega_sigma_and_theta.tangent.template head<3>();
155  Scalar sigma = omega_sigma_and_theta.tangent[3];
156  Matrix3<Scalar> const Omega = SO3<Scalar>::hat(omega);
157  Matrix3<Scalar> const W_inv = details::calcWInv<Scalar, 3>(
158  Omega, omega_sigma_and_theta.theta, sigma, scale());
159 
160  res.segment(0, 3) = W_inv * translation();
161  res.segment(3, 3) = omega;
162  res[6] = sigma;
163  return res;
164  }

474  SOPHUS_FUNC static Sim3<Scalar> exp(Tangent const& a) {
475  // For the derivation of the exponential map of Sim(3) see
476  // H. Strasdat, "Local Accuracy and Global Consistency for Efficient Visual
477  // SLAM", PhD thesis, 2012.
478  // http:///hauke.strasdat.net/files/strasdat_thesis_2012.pdf (A.5, pp. 186)
479  Vector3<Scalar> const upsilon = a.segment(0, 3);
480  Vector3<Scalar> const omega = a.segment(3, 3);
481  Scalar const sigma = a[6];
482  Scalar theta;
483  RxSO3<Scalar> rxso3 =
484  RxSO3<Scalar>::expAndTheta(a.template tail<4>(), &theta);
485  Matrix3<Scalar> const Omega = SO3<Scalar>::hat(omega);
486 
487  Matrix3<Scalar> const W = details::calcW<Scalar, 3>(Omega, theta, sigma);
488  return Sim3<Scalar>(rxso3, W * upsilon);
489  }

573  SOPHUS_FUNC static Tangent lieBracket(Tangent const& a, Tangent const& b) {
574  Vector3<Scalar> const upsilon1 = a.template head<3>();
575  Vector3<Scalar> const upsilon2 = b.template head<3>();
576  Vector3<Scalar> const omega1 = a.template segment<3>(3);
577  Vector3<Scalar> const omega2 = b.template segment<3>(3);
578  Scalar sigma1 = a[6];
579  Scalar sigma2 = b[6];
580 
581  Tangent res;
582  res.template head<3>() = SO3<Scalar>::hat(omega1) * upsilon2 +
583  SO3<Scalar>::hat(upsilon1) * omega2 +
584  sigma1 * upsilon2 - sigma2 * upsilon1;
585  res.template segment<3>(3) = omega1.cross(omega2);
586  res[6] = Scalar(0);
587 
588  return res;
589  }
