46 def transform44(l):
47  """
48  Generate a 4x4 homogeneous transformation matrix from a 3D point and unit quaternion.
49 
50  Input:
51  l -- tuple consisting of (stamp,tx,ty,tz,qx,qy,qz,qw) where
52  (tx,ty,tz) is the 3D position and (qx,qy,qz,qw) is the unit quaternion.
53 
54  Output:
55  matrix -- 4x4 homogeneous transformation matrix
56  """
57  t = l[1:4]
58  q = numpy.array(l[4:8], dtype=numpy.float64, copy=True)
59  nq = numpy.dot(q, q)
60  if nq < _EPS:
61  return numpy.array((
62  ( 1.0, 0.0, 0.0, t[0])
63  ( 0.0, 1.0, 0.0, t[1])
64  ( 0.0, 0.0, 1.0, t[2])
65  ( 0.0, 0.0, 0.0, 1.0)
66  ), dtype=numpy.float64)
67  q *= numpy.sqrt(2.0 / nq)
68  q = numpy.outer(q, q)
69  return numpy.array((
70  (1.0-q[1, 1]-q[2, 2], q[0, 1]-q[2, 3], q[0, 2]+q[1, 3], t[0]),
71  ( q[0, 1]+q[2, 3], 1.0-q[0, 0]-q[2, 2], q[1, 2]-q[0, 3], t[1]),
72  ( q[0, 2]-q[1, 3], q[1, 2]+q[0, 3], 1.0-q[0, 0]-q[1, 1], t[2]),
73  ( 0.0, 0.0, 0.0, 1.0)
74  ), dtype=numpy.float64)

76 def read_trajectory(filename, matrix=True):
77  """
78  Read a trajectory from a text file.
79 
80  Input:
81  filename -- file to be read
82  matrix -- convert poses to 4x4 matrices
83 
84  Output:
85  dictionary of stamped 3D poses
86  """
87  file = open(filename)
88  data = file.read()
89  lines = data.replace(","," ").replace("\t"," ").split("\n")
90  list = [[float(v.strip()) for v in line.split(" ") if v.strip()!=""] for line in lines if len(line)>0 and line[0]!="#"]
91  list_ok = []
92  for i,l in enumerate(list):
93  if l[4:8]==[0,0,0,0]:
94  continue
95  isnan = False
96  for v in l:
97  if numpy.isnan(v):
98  isnan = True
99  break
100  if isnan:
101  sys.stderr.write("Warning: line %d of file '%s' has NaNs, skipping line\n"%(i,filename))
102  continue
103  list_ok.append(l)
104  if matrix :
105  traj = dict([(l[0],transform44(l[0:])) for l in list_ok])
106  else:
107  traj = dict([(l[0],l[1:8]) for l in list_ok])
108  return traj

110 def find_closest_index(L,t):
111  """
112  Find the index of the closest value in a list.
113 
114  Input:
115  L -- the list
116  t -- value to be found
117 
118  Output:
119  index of the closest element
120  """
121  beginning = 0
122  difference = abs(L[0] - t)
123  best = 0
124  end = len(L)
125  while beginning < end:
126  middle = int((end+beginning)/2)
127  if abs(L[middle] - t) < difference:
128  difference = abs(L[middle] - t)
129  best = middle
130  if t == L[middle]:
131  return middle
132  elif L[middle] > t:
133  end = middle
134  else:
135  beginning = middle + 1
136  return best

138 def ominus(a,b):
139  """
140  Compute the relative 3D transformation between a and b.
141 
142  Input:
143  a -- first pose (homogeneous 4x4 matrix)
144  b -- second pose (homogeneous 4x4 matrix)
145 
146  Output:
147  Relative 3D transformation from a to b.
148  """
149  return numpy.dot(numpy.linalg.inv(a),b)

151 def scale(a,scalar):
152  """
153  Scale the translational components of a 4x4 homogeneous matrix by a scale factor.
154  """
155  return numpy.array(
156  [[a[0,0], a[0,1], a[0,2], a[0,3]*scalar],
157  [a[1,0], a[1,1], a[1,2], a[1,3]*scalar],
158  [a[2,0], a[2,1], a[2,2], a[2,3]*scalar],
159  [a[3,0], a[3,1], a[3,2], a[3,3]]]
160  )

162 def compute_distance(transform):
163  """
164  Compute the distance of the translational component of a 4x4 homogeneous matrix.
165  """
166  return numpy.linalg.norm(transform[0:3,3])

168 def compute_angle(transform):
169  """
170  Compute the rotation angle from a 4x4 homogeneous matrix.
171  """
172  # an invitation to 3-d vision, p 27
173  return numpy.arccos( min(1,max(-1, (numpy.trace(transform[0:3,0:3]) - 1)/2) ))

175 def distances_along_trajectory(traj):
176  """
177  Compute the translational distances along a trajectory.
178  """
179  keys = traj.keys()
180  keys.sort()
181  motion = [ominus(traj[keys[i+1]],traj[keys[i]]) for i in range(len(keys)-1)]
182  distances = [0]
183  sum = 0
184  for t in motion:
185  sum += compute_distance(t)
186  distances.append(sum)
187  return distances

204 def evaluate_trajectory(traj_gt,traj_est,param_max_pairs=10000,param_fixed_delta=False,param_delta=1.00,param_delta_unit="s",param_offset=0.00,param_scale=1.00):
205  """
206  Compute the relative pose error between two trajectories.
207 
208  Input:
209  traj_gt -- the first trajectory (ground truth)
210  traj_est -- the second trajectory (estimated trajectory)
211  param_max_pairs -- number of relative poses to be evaluated
212  param_fixed_delta -- false: evaluate over all possible pairs
213  true: only evaluate over pairs with a given distance (delta)
214  param_delta -- distance between the evaluated pairs
215  param_delta_unit -- unit for comparison:
216  "s": seconds
217  "m": meters
218  "rad": radians
219  "deg": degrees
220  "f": frames
221  param_offset -- time offset between two trajectories (to model the delay)
222  param_scale -- scale to be applied to the second trajectory
223 
224  Output:
225  list of compared poses and the resulting translation and rotation error
226  """
227  stamps_gt = list(traj_gt.keys())
228  stamps_est = list(traj_est.keys())
229  stamps_gt.sort()
230  stamps_est.sort()
231 
232  stamps_est_return = []
233  for t_est in stamps_est:
234  t_gt = stamps_gt[find_closest_index(stamps_gt,t_est + param_offset)]
235  t_est_return = stamps_est[find_closest_index(stamps_est,t_gt - param_offset)]
236  t_gt_return = stamps_gt[find_closest_index(stamps_gt,t_est_return + param_offset)]
237  if not t_est_return in stamps_est_return:
238  stamps_est_return.append(t_est_return)
239  if(len(stamps_est_return)<2):
240  raise Exception("Number of overlap in the timestamps is too small. Did you run the evaluation on the right files?")
241 
242  if param_delta_unit=="s":
243  index_est = list(traj_est.keys())
244  index_est.sort()
245  elif param_delta_unit=="m":
246  index_est = distances_along_trajectory(traj_est)
247  elif param_delta_unit=="rad":
248  index_est = rotations_along_trajectory(traj_est,1)
249  elif param_delta_unit=="deg":
250  index_est = rotations_along_trajectory(traj_est,180/numpy.pi)
251  elif param_delta_unit=="f":
252  index_est = range(len(traj_est))
253  else:
254  raise Exception("Unknown unit for delta: '%s'"%param_delta_unit)
255 
256  if not param_fixed_delta:
257  if(param_max_pairs==0 or len(traj_est)<numpy.sqrt(param_max_pairs)):
258  pairs = [(i,j) for i in range(len(traj_est)) for j in range(len(traj_est))]
259  else:
260  pairs = [(random.randint(0,len(traj_est)-1),random.randint(0,len(traj_est)-1)) for i in range(param_max_pairs)]
261  else:
262  pairs = []
263  for i in range(len(traj_est)):
264  j = find_closest_index(index_est,index_est[i] + param_delta)
265  if j!=len(traj_est)-1:
266  pairs.append((i,j))
267  if(param_max_pairs!=0 and len(pairs)>param_max_pairs):
268  pairs = random.sample(pairs,param_max_pairs)
269 
270  gt_interval = numpy.median([s-t for s,t in zip(stamps_gt[1:],stamps_gt[:-1])])
271  gt_max_time_difference = 2*gt_interval
272 
273  result = []
274  for i,j in pairs:
275  stamp_est_0 = stamps_est[i]
276  stamp_est_1 = stamps_est[j]
277 
278  stamp_gt_0 = stamps_gt[ find_closest_index(stamps_gt,stamp_est_0 + param_offset) ]
279  stamp_gt_1 = stamps_gt[ find_closest_index(stamps_gt,stamp_est_1 + param_offset) ]
280 
281  if(abs(stamp_gt_0 - (stamp_est_0 + param_offset)) > gt_max_time_difference or
282  abs(stamp_gt_1 - (stamp_est_1 + param_offset)) > gt_max_time_difference):
283  continue
284 
285  error44 = ominus( scale(
286  ominus( traj_est[stamp_est_1], traj_est[stamp_est_0] ),param_scale),
287  ominus( traj_gt[stamp_gt_1], traj_gt[stamp_gt_0] ) )
288 
289  trans = compute_distance(error44)
290  rot = compute_angle(error44)
291 
292  result.append([stamp_est_0,stamp_est_1,stamp_gt_0,stamp_gt_1,trans,rot])
293 
294  if len(result)<2:
295  raise Exception("Couldn't find matching timestamp pairs between groundtruth and estimated trajectory!")
296 
297  return result
