271 std::vector<float> Converter::toEuler(const cv::Mat &R)
272 {
273  assert(isRotationMatrix(R));
274  float sy = sqrt(R.at<float>(0,0) * R.at<float>(0,0) + R.at<float>(1,0) * R.at<float>(1,0) );
275 
276  bool singular = sy < 1e-6; // If
277 
278  float x, y, z;
279  if (!singular)
280  {
281  x = atan2(R.at<float>(2,1) , R.at<float>(2,2));
282  y = atan2(-R.at<float>(2,0), sy);
283  z = atan2(R.at<float>(1,0), R.at<float>(0,0));
284  }
285  else
286  {
287  x = atan2(-R.at<float>(1,2), R.at<float>(1,1));
288  y = atan2(-R.at<float>(2,0), sy);
289  z = 0;
290  }
291 
292  std::vector<float> v_euler(3);
293  v_euler[0] = x;
294  v_euler[1] = y;
295  v_euler[2] = z;
