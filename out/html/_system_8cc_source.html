569 void System::SaveTrajectoryTUM(const string &filename)
570 {
571  cout << endl << "Saving camera trajectory to " << filename << " ..." << endl;
572  if(mSensor==MONOCULAR)
573  {
574  cerr << "ERROR: SaveTrajectoryTUM cannot be used for monocular." << endl;
575  return;
576  }
577 
578  vector<KeyFrame*> vpKFs = mpAtlas->GetAllKeyFrames();
579  sort(vpKFs.begin(),vpKFs.end(),KeyFrame::lId);
580 
581  // Transform all keyframes so that the first keyframe is at the origin.
582  // After a loop closure the first keyframe might not be at the origin.
583  Sophus::SE3f Two = vpKFs[0]->GetPoseInverse();
584 
585  ofstream f;
586  f.open(filename.c_str());
587  f << fixed;
588 
589  // Frame pose is stored relative to its reference keyframe (which is optimized by BA and pose graph).
590  // We need to get first the keyframe pose and then concatenate the relative transformation.
591  // Frames not localized (tracking failure) are not saved.
592 
593  // For each frame we have a reference keyframe (lRit), the timestamp (lT) and a flag
594  // which is true when tracking failed (lbL).
595  list<ORB_SLAM3::KeyFrame*>::iterator lRit = mpTracker->mlpReferences.begin();
596  list<double>::iterator lT = mpTracker->mlFrameTimes.begin();
597  list<bool>::iterator lbL = mpTracker->mlbLost.begin();
598  for(list<Sophus::SE3f>::iterator lit=mpTracker->mlRelativeFramePoses.begin(),
599  lend=mpTracker->mlRelativeFramePoses.end();lit!=lend;lit++, lRit++, lT++, lbL++)
600  {
601  if(*lbL)
602  continue;
603 
604  KeyFrame* pKF = *lRit;
605 
606  Sophus::SE3f Trw;
607 
608  // If the reference keyframe was culled, traverse the spanning tree to get a suitable keyframe.
609  while(pKF->isBad())
610  {
611  Trw = Trw * pKF->mTcp;
612  pKF = pKF->GetParent();
613  }
614 
615  Trw = Trw * pKF->GetPose() * Two;
616 
617  Sophus::SE3f Tcw = (*lit) * Trw;
618  Sophus::SE3f Twc = Tcw.inverse();
619 
620  Eigen::Vector3f twc = Twc.translation();
621  Eigen::Quaternionf q = Twc.unit_quaternion();
622 
623  f << setprecision(6) << *lT << " " << setprecision(9) << twc(0) << " " << twc(1) << " " << twc(2) << " " << q.x() << " " << q.y() << " " << q.z() << " " << q.w() << endl;
624  }
625  f.close();
626  // cout << endl << "trajectory saved!" << endl;
