116  cv::Point3f KannalaBrandt8::unproject(const cv::Point2f &p2D) {
117  //Use Newthon method to solve for theta with good precision (err ~ e-6)
118  cv::Point2f pw((p2D.x - mvParameters[2]) / mvParameters[0], (p2D.y - mvParameters[3]) / mvParameters[1]);
119  float scale = 1.f;
120  float theta_d = sqrtf(pw.x * pw.x + pw.y * pw.y);
121  theta_d = fminf(fmaxf(-CV_PI / 2.f, theta_d), CV_PI / 2.f);
122 
123  if (theta_d > 1e-8) {
124  //Compensate distortion iteratively
125  float theta = theta_d;
126 
127  for (int j = 0; j < 10; j++) {
128  float theta2 = theta * theta, theta4 = theta2 * theta2, theta6 = theta4 * theta2, theta8 =
129  theta4 * theta4;
130  float k0_theta2 = mvParameters[4] * theta2, k1_theta4 = mvParameters[5] * theta4;
131  float k2_theta6 = mvParameters[6] * theta6, k3_theta8 = mvParameters[7] * theta8;
132  float theta_fix = (theta * (1 + k0_theta2 + k1_theta4 + k2_theta6 + k3_theta8) - theta_d) /
133  (1 + 3 * k0_theta2 + 5 * k1_theta4 + 7 * k2_theta6 + 9 * k3_theta8);
134  theta = theta - theta_fix;
135  if (fabsf(theta_fix) < precision)
136  break;
137  }
138  //scale = theta - theta_d;
139  scale = std::tan(theta) / theta_d;
140  }
141 
142  return cv::Point3f(pw.x * scale, pw.y * scale, 1.f);
143  }

145  Eigen::Matrix<double, 2, 3> KannalaBrandt8::projectJac(const Eigen::Vector3d &v3D) {
146  double x2 = v3D[0] * v3D[0], y2 = v3D[1] * v3D[1], z2 = v3D[2] * v3D[2];
147  double r2 = x2 + y2;
148  double r = sqrt(r2);
149  double r3 = r2 * r;
150  double theta = atan2(r, v3D[2]);
151 
152  double theta2 = theta * theta, theta3 = theta2 * theta;
153  double theta4 = theta2 * theta2, theta5 = theta4 * theta;
154  double theta6 = theta2 * theta4, theta7 = theta6 * theta;
155  double theta8 = theta4 * theta4, theta9 = theta8 * theta;
156 
157  double f = theta + theta3 * mvParameters[4] + theta5 * mvParameters[5] + theta7 * mvParameters[6] +
158  theta9 * mvParameters[7];
159  double fd = 1 + 3 * mvParameters[4] * theta2 + 5 * mvParameters[5] * theta4 + 7 * mvParameters[6] * theta6 +
160  9 * mvParameters[7] * theta8;
161 
162  Eigen::Matrix<double, 2, 3> JacGood;
163  JacGood(0, 0) = mvParameters[0] * (fd * v3D[2] * x2 / (r2 * (r2 + z2)) + f * y2 / r3);
164  JacGood(1, 0) =
165  mvParameters[1] * (fd * v3D[2] * v3D[1] * v3D[0] / (r2 * (r2 + z2)) - f * v3D[1] * v3D[0] / r3);
166 
167  JacGood(0, 1) =
168  mvParameters[0] * (fd * v3D[2] * v3D[1] * v3D[0] / (r2 * (r2 + z2)) - f * v3D[1] * v3D[0] / r3);
169  JacGood(1, 1) = mvParameters[1] * (fd * v3D[2] * y2 / (r2 * (r2 + z2)) + f * x2 / r3);
170 
171  JacGood(0, 2) = -mvParameters[0] * fd * v3D[0] / (r2 + z2);
172  JacGood(1, 2) = -mvParameters[1] * fd * v3D[1] / (r2 + z2);
173 
174  return JacGood;
175  }

222  bool KannalaBrandt8::matchAndtriangulate(const cv::KeyPoint& kp1, const cv::KeyPoint& kp2, GeometricCamera* pOther,
223  Sophus::SE3f& Tcw1, Sophus::SE3f& Tcw2,
224  const float sigmaLevel1, const float sigmaLevel2,
225  Eigen::Vector3f& x3Dtriangulated){
226  Eigen::Matrix<float,3,4> eigTcw1 = Tcw1.matrix3x4();
227  Eigen::Matrix3f Rcw1 = eigTcw1.block<3,3>(0,0);
228  Eigen::Matrix3f Rwc1 = Rcw1.transpose();
229  Eigen::Matrix<float,3,4> eigTcw2 = Tcw2.matrix3x4();
230  Eigen::Matrix3f Rcw2 = eigTcw2.block<3,3>(0,0);
231  Eigen::Matrix3f Rwc2 = Rcw2.transpose();
232 
233  cv::Point3f ray1c = this->unproject(kp1.pt);
234  cv::Point3f ray2c = pOther->unproject(kp2.pt);
235 
236  Eigen::Vector3f r1(ray1c.x, ray1c.y, ray1c.z);
237  Eigen::Vector3f r2(ray2c.x, ray2c.y, ray2c.z);
238 
239  //Check parallax between rays
240  Eigen::Vector3f ray1 = Rwc1 * r1;
241  Eigen::Vector3f ray2 = Rwc2 * r2;
242 
243  const float cosParallaxRays = ray1.dot(ray2)/(ray1.norm() * ray2.norm());
244 
245  //If parallax is lower than 0.9998, reject this match
246  if(cosParallaxRays > 0.9998){
247  return false;
248  }
249 
250  //Parallax is good, so we try to triangulate
251  cv::Point2f p11,p22;
252 
253  p11.x = ray1c.x;
254  p11.y = ray1c.y;
255 
256  p22.x = ray2c.x;
257  p22.y = ray2c.y;
258 
259  Eigen::Vector3f x3D;
260 
261  Triangulate(p11,p22,eigTcw1,eigTcw2,x3D);
262 
263  //Check triangulation in front of cameras
264  float z1 = Rcw1.row(2).dot(x3D)+Tcw1.translation()(2);
265  if(z1<=0){ //Point is not in front of the first camera
266  return false;
267  }
268 
269 
270  float z2 = Rcw2.row(2).dot(x3D)+Tcw2.translation()(2);
271  if(z2<=0){ //Point is not in front of the first camera
272  return false;
273  }
274 
275  //Check reprojection error in first keyframe
276  // -Transform point into camera reference system
277  Eigen::Vector3f x3D1 = Rcw1 * x3D + Tcw1.translation();
278  Eigen::Vector2f uv1 = this->project(x3D1);
279 
280  float errX1 = uv1(0) - kp1.pt.x;
281  float errY1 = uv1(1) - kp1.pt.y;
282 
283  if((errX1*errX1+errY1*errY1)>5.991*sigmaLevel1){ //Reprojection error is high
284  return false;
285  }
286 
287  //Check reprojection error in second keyframe;
288  // -Transform point into camera reference system
289  Eigen::Vector3f x3D2 = Rcw2 * x3D + Tcw2.translation(); // avoid using q
290  Eigen::Vector2f uv2 = pOther->project(x3D2);
291 
292  float errX2 = uv2(0) - kp2.pt.x;
293  float errY2 = uv2(1) - kp2.pt.y;
294 
295  if((errX2*errX2+errY2*errY2)>5.991*sigmaLevel2){ //Reprojection error is high
296  return false;
297  }
298 
299  //Since parallax is big enough and reprojection errors are low, this pair of points
300  //can be considered as a match
301  x3Dtriangulated = x3D;
302 
303  return true;
304  }
