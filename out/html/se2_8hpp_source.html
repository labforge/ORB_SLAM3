476  SOPHUS_FUNC static Sophus::Matrix<Scalar, num_parameters, DoF> Dx_exp_x(
477  Tangent const& upsilon_theta) {
478  using std::abs;
479  using std::cos;
480  using std::pow;
481  using std::sin;
482  Sophus::Matrix<Scalar, num_parameters, DoF> J;
483  Sophus::Vector<Scalar, 2> upsilon = upsilon_theta.template head<2>();
484  Scalar theta = upsilon_theta[2];
485 
486  if (abs(theta) < Constants<Scalar>::epsilon()) {
487  Scalar const o(0);
488  Scalar const i(1);
489 
490  // clang-format off
491  J << o, o, o, o, o, i, i, o, -Scalar(0.5) * upsilon[1], o, i,
492  Scalar(0.5) * upsilon[0];
493  // clang-format on
494  return J;
495  }
496 
497  Scalar const c0 = sin(theta);
498  Scalar const c1 = cos(theta);
499  Scalar const c2 = 1.0 / theta;
500  Scalar const c3 = c0 * c2;
501  Scalar const c4 = -c1 + Scalar(1);
502  Scalar const c5 = c2 * c4;
503  Scalar const c6 = c1 * c2;
504  Scalar const c7 = pow(theta, -2);
505  Scalar const c8 = c0 * c7;
506  Scalar const c9 = c4 * c7;
507 
508  Scalar const o = Scalar(0);
509  J(0, 0) = o;
510  J(0, 1) = o;
511  J(0, 2) = -c0;
512  J(1, 0) = o;
513  J(1, 1) = o;
514  J(1, 2) = c1;
515  J(2, 0) = c3;
516  J(2, 1) = -c5;
517  J(2, 2) =
518  -c3 * upsilon[1] + c6 * upsilon[0] - c8 * upsilon[0] + c9 * upsilon[1];
519  J(3, 0) = c5;
520  J(3, 1) = c3;
521  J(3, 2) =
522  c3 * upsilon[0] + c6 * upsilon[1] - c8 * upsilon[1] - c9 * upsilon[0];

524  }

558  SOPHUS_FUNC static SE2<Scalar> exp(Tangent const& a) {
559  Scalar theta = a[2];
560  SO2<Scalar> so2 = SO2<Scalar>::exp(theta);
561  Scalar sin_theta_by_theta;
562  Scalar one_minus_cos_theta_by_theta;
563  using std::abs;
564 
565  if (abs(theta) < Constants<Scalar>::epsilon()) {
566  Scalar theta_sq = theta * theta;
567  sin_theta_by_theta = Scalar(1.) - Scalar(1. / 6.) * theta_sq;
568  one_minus_cos_theta_by_theta =
569  Scalar(0.5) * theta - Scalar(1. / 24.) * theta * theta_sq;
570  } else {
571  sin_theta_by_theta = so2.unit_complex().y() / theta;
572  one_minus_cos_theta_by_theta =
573  (Scalar(1.) - so2.unit_complex().x()) / theta;
574  }
575  Vector2<Scalar> trans(
576  sin_theta_by_theta * a[0] - one_minus_cos_theta_by_theta * a[1],
577  one_minus_cos_theta_by_theta * a[0] + sin_theta_by_theta * a[1]);
578  return SE2<Scalar>(so2, trans);
579  }
