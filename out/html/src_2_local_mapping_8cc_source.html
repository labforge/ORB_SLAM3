936  {
937  //cout<<"redundant_th Option3"<<endl;
938  redundant_th = 0.5;

942  int count=0;

954  }

961  {

966  continue;

968

974  {

990  : (i < pKF -> NLeft) ? pKF -> mvKeys[i].octave

993  int nObs=0;

995  {

997  if(pKFi==pKF)

1000  int leftIndex = get<0>(indexes), rightIndex = get<1>(indexes);
1001  int scaleLeveli = -1;
1002  if(pKFi -> NLeft == -1)
1003  scaleLeveli = pKFi->mvKeysUn[leftIndex].octave;

1017  nObs++;

1019  break;
1020  }
1021  }
1022  if(nObs>thObs)
1023  {
1024  nRedundantObservations++;
1025  }
1026  }
1027  }
1028  }
1029  }
1030 
1031  if(nRedundantObservations>redundant_th*nMPs)

1033  if (mbInertial)
1034  {
1035  if (mpAtlas->KeyFramesInMap()<=Nd)
1036  continue;

1038  if(pKF->mnId>(mpCurrentKeyFrame->mnId-2))

1253  {
