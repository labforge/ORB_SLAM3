37 cv::Mat ExpSO3(const float &x, const float &y, const float &z)
38 {
39  cv::Mat I = cv::Mat::eye(3,3,CV_32F);
40  const float d2 = x*x+y*y+z*z;
41  const float d = sqrt(d2);
42  cv::Mat W = (cv::Mat_<float>(3,3) << 0, -z, y,
43  z, 0, -x,
44  -y, x, 0);
45  if(d<eps)
46  return (I + W + 0.5f*W*W);
47  else
48  return (I + W*sin(d)/d + W*W*(1.0f-cos(d))/d2);
49 }

435  //RANSAC

437  {
438  vAvailableIndices = vAllIndices;
439 
440  cv::Mat A(3,4,CV_32F);

447 
448  int idx = vAvailableIndices[randi];

459  const float a = vt.at<float>(3,0);

462  const float d = vt.at<float>(3,3);

519 void Plane::Recompute()
520 {
521  const int N = mvMPs.size();
522 
523  // Recompute plane with all points
524  cv::Mat A = cv::Mat(N,4,CV_32F);
525  A.col(3) = cv::Mat::ones(N,1,CV_32F);
526 
527  o = cv::Mat::zeros(3,1,CV_32F);
528 
529  int nPoints = 0;
530  for(int i=0; i<N; i++)
531  {
532  MapPoint* pMP = mvMPs[i];
533  if(!pMP->isBad())
534  {
535  cv::Mat Xw;
536  cv::eigen2cv(pMP->GetWorldPos(), Xw);
537  o+=Xw;
538  A.row(nPoints).colRange(0,3) = Xw.t();
539  nPoints++;
540  }
541  }
542  A.resize(nPoints);
543 
544  cv::Mat u,w,vt;
545  cv::SVDecomp(A,w,u,vt,cv::SVD::MODIFY_A | cv::SVD::FULL_UV);
546 
547  float a = vt.at<float>(3,0);
548  float b = vt.at<float>(3,1);
549  float c = vt.at<float>(3,2);
550 
551  o = o*(1.0f/nPoints);
552  const float f = 1.0f/sqrt(a*a+b*b+c*c);
553 
554  // Compute XC just the first time
555  if(XC.empty())
556  {
557  cv::Mat Oc = -mTcw.colRange(0,3).rowRange(0,3).t()*mTcw.rowRange(0,3).col(3);
558  XC = Oc-o;
559  }
560 
561  if((XC.at<float>(0)*a+XC.at<float>(1)*b+XC.at<float>(2)*c)>0)
562  {
563  a=-a;
564  b=-b;
565  c=-c;
566  }
567 
568  const float nx = a*f;
569  const float ny = b*f;
570  const float nz = c*f;
571 
572  n = (cv::Mat_<float>(3,1)<<nx,ny,nz);
573 
574  cv::Mat up = (cv::Mat_<float>(3,1) << 0.0f, 1.0f, 0.0f);
575 
576  cv::Mat v = up.cross(n);
577  const float sa = cv::norm(v);
578  const float ca = up.dot(n);
579  const float ang = atan2(sa,ca);
580  Tpw = cv::Mat::eye(4,4,CV_32F);
581 
582 
583  Tpw.rowRange(0,3).colRange(0,3) = ExpSO3(v*ang/sa)*ExpSO3(up*rang);
584  o.copyTo(Tpw.col(3).rowRange(0,3));
