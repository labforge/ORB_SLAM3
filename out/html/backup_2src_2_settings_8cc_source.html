485  void Settings::precomputeRectificationMaps() {
486  //Precompute rectification maps, new calibrations, ...
487  cv::Mat K1 = static_cast<Pinhole*>(calibration1_)->toK();
488  K1.convertTo(K1,CV_64F);
489  cv::Mat K2 = static_cast<Pinhole*>(calibration2_)->toK();
490  K2.convertTo(K2,CV_64F);
491 
492  cv::Mat cvTlr;
493  cv::eigen2cv(Tlr_.inverse().matrix3x4(),cvTlr);
494  cv::Mat R12 = cvTlr.rowRange(0,3).colRange(0,3);
495  R12.convertTo(R12,CV_64F);
496  cv::Mat t12 = cvTlr.rowRange(0,3).col(3);
497  t12.convertTo(t12,CV_64F);
498 
499  cv::Mat R_r1_u1, R_r2_u2;
500  cv::Mat P1, P2, Q;
501 
502  cv::stereoRectify(K1,camera1DistortionCoef(),K2,camera2DistortionCoef(),newImSize_,
503  R12, t12,
504  R_r1_u1,R_r2_u2,P1,P2,Q,
505  cv::CALIB_ZERO_DISPARITY,-1,newImSize_);
506  cv::initUndistortRectifyMap(K1, camera1DistortionCoef(), R_r1_u1, P1.rowRange(0, 3).colRange(0, 3),
507  newImSize_, CV_32F, M1l_, M2l_);
508  cv::initUndistortRectifyMap(K2, camera2DistortionCoef(), R_r2_u2, P2.rowRange(0, 3).colRange(0, 3),
509  newImSize_, CV_32F, M1r_, M2r_);
510 
511  //Update calibration
512  calibration1_->setParameter(P1.at<double>(0,0), 0);
513  calibration1_->setParameter(P1.at<double>(1,1), 1);
514  calibration1_->setParameter(P1.at<double>(0,2), 2);
515  calibration1_->setParameter(P1.at<double>(1,2), 3);
516 
517  //Update bf
518  bf_ = b_ * P1.at<double>(0,0);
519 
520  //Update relative pose between camera 1 and IMU if necessary
521  if(sensor_ == System::IMU_STEREO){
522  Eigen::Matrix3f eigenR_r1_u1;
523  cv::cv2eigen(R_r1_u1,eigenR_r1_u1);
524  Sophus::SE3f T_r1_u1(eigenR_r1_u1,Eigen::Vector3f::Zero());
525  Tbc_ = Tbc_ * T_r1_u1.inverse();
526  }
527  }
