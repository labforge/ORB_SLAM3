143 void EdgeSE3ProjectXYZ::linearizeOplus() {
144  VertexSE3Expmap * vj = static_cast<VertexSE3Expmap *>(_vertices[1]);
145  SE3Quat T(vj->estimate());
146  VertexSBAPointXYZ* vi = static_cast<VertexSBAPointXYZ*>(_vertices[0]);
147  Vector3d xyz = vi->estimate();
148  Vector3d xyz_trans = T.map(xyz);
149 
150  double x = xyz_trans[0];
151  double y = xyz_trans[1];
152  double z = xyz_trans[2];
153  double z_2 = z*z;
154 
155  Matrix<double,2,3> tmp;
156  tmp(0,0) = fx;
157  tmp(0,1) = 0;
158  tmp(0,2) = -x/z*fx;
159 
160  tmp(1,0) = 0;
161  tmp(1,1) = fy;
162  tmp(1,2) = -y/z*fy;
163 
164  _jacobianOplusXi = -1./z * tmp * T.rotation().toRotationMatrix();
165 
166  _jacobianOplusXj(0,0) = x*y/z_2 *fx;
167  _jacobianOplusXj(0,1) = -(1+(x*x/z_2)) *fx;
168  _jacobianOplusXj(0,2) = y/z *fx;
169  _jacobianOplusXj(0,3) = -1./z *fx;
170  _jacobianOplusXj(0,4) = 0;
171  _jacobianOplusXj(0,5) = x/z_2 *fx;
172 
173  _jacobianOplusXj(1,0) = (1+y*y/z_2) *fy;
174  _jacobianOplusXj(1,1) = -x*y/z_2 *fy;
175  _jacobianOplusXj(1,2) = -x/z *fy;
176  _jacobianOplusXj(1,3) = 0;
177  _jacobianOplusXj(1,4) = -1./z *fy;
178  _jacobianOplusXj(1,5) = y/z_2 *fy;
179 }

248  _jacobianOplusXi(1,2) = -fy*R(1,2)/z+fy*y*R(2,2)/z_2;

251  _jacobianOplusXi(2,1) = _jacobianOplusXi(0,1)-bf*R(2,1)/z_2;
252  _jacobianOplusXi(2,2) = _jacobianOplusXi(0,2)-bf*R(2,2)/z_2;
253 
254  _jacobianOplusXj(0,0) = x*y/z_2 *fx;
