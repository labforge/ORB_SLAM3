40 {

44  const float d = sqrt(d2);

47  Eigen::Matrix3f W = Sophus::SO3f::hat(v);

177 void Preintegrated::IntegrateNewMeasurement(const Eigen::Vector3f &acceleration, const Eigen::Vector3f &angVel, const float &dt)
178 {
179  mvMeasurements.push_back(integrable(acceleration,angVel,dt));
180 
181  // Position is updated firstly, as it depends on previously computed velocity and rotation.
182  // Velocity is updated secondly, as it depends on previously computed rotation.
183  // Rotation is the last to be updated.
184 
185  //Matrices to compute covariance
186  Eigen::Matrix<float,9,9> A;
187  A.setIdentity();
188  Eigen::Matrix<float,9,6> B;
189  B.setZero();
190 
191  Eigen::Vector3f acc, accW;
192  acc << acceleration(0)-b.bax, acceleration(1)-b.bay, acceleration(2)-b.baz;
193  accW << angVel(0)-b.bwx, angVel(1)-b.bwy, angVel(2)-b.bwz;
194 
195  avgA = (dT*avgA + dR*acc*dt)/(dT+dt);
196  avgW = (dT*avgW + accW*dt)/(dT+dt);
197 
198  // Update delta position dP and velocity dV (rely on no-updated delta rotation)
199  dP = dP + dV*dt + 0.5f*dR*acc*dt*dt;
200  dV = dV + dR*acc*dt;
201 
202  // Compute velocity and position parts of matrices A and B (rely on non-updated delta rotation)
203  Eigen::Matrix<float,3,3> Wacc = Sophus::SO3f::hat(acc);
204 
205  A.block<3,3>(3,0) = -dR*dt*Wacc;
206  A.block<3,3>(6,0) = -0.5f*dR*dt*dt*Wacc;
207  A.block<3,3>(6,3) = Eigen::DiagonalMatrix<float,3>(dt, dt, dt);
208  B.block<3,3>(3,3) = dR*dt;
209  B.block<3,3>(6,3) = 0.5f*dR*dt*dt;
210 
211 
212  // Update position and velocity jacobians wrt bias correction
213  JPa = JPa + JVa*dt -0.5f*dR*dt*dt;
214  JPg = JPg + JVg*dt -0.5f*dR*dt*dt*Wacc*JRg;
215  JVa = JVa - dR*dt;
216  JVg = JVg - dR*dt*Wacc*JRg;
217 
218  // Update delta rotation
219  IntegratedRotation dRi(angVel,b,dt);
220  dR = NormalizeRotation(dR*dRi.deltaR);
221 
222  // Compute rotation parts of matrices A and B
223  A.block<3,3>(0,0) = dRi.deltaR.transpose();
224  B.block<3,3>(0,0) = dRi.rightJ*dt;
225 
226  // Update covariance
227  C.block<9,9>(0,0) = A * C.block<9,9>(0,0) * A.transpose() + B*Nga*B.transpose();
228  C.block<6,6>(9,9) += NgaWalk;
229 
230  // Update rotation jacobian wrt bias correction
231  JRg = dRi.deltaR.transpose()*JRg - dRi.rightJ*dt;
232 
233  // Total integrated time
234  dT += dt;
235 }
