180  enable_if_t<std::is_floating_point<S>::value, bool> testSampleUniformSymmetry() {
181  bool passed = true;
182  std::default_random_engine generator(0);
183 
184  // A non-rigorous test for checking that our sampleUniform() function is
185  // giving us symmetric results
186  //
187  // We (a) split the output space in half, (b) apply a series of random
188  // rotations to a point, (c) check which half of the output space each
189  // transformed point ends up, and then (d) apply a standard "coin toss"
190  // chi-square test
191 
192  for (size_t trial = 0; trial < 5; trial++) {
193  std::normal_distribution<Scalar> normal(0, 10);
194 
195  // Pick a random plane to split the output space by
196  Point plane_normal(normal(generator), normal(generator),
197  normal(generator));
198  plane_normal /= plane_normal.norm();
199 
200  // Pick a random point to be rotated
201  Point input_point(normal(generator), normal(generator),
202  normal(generator));
203  input_point /= input_point.norm();
204 
205  // Randomly rotate points and track # that land on each side of plane
206  size_t positive_count = 0;
207  size_t negative_count = 0;
208  size_t samples = 5000;
209  for (size_t i = 0; i < samples; ++i) {
210  SO3Type R = SO3Type::sampleUniform(generator);
211  if (plane_normal.dot(R * input_point) > 0)
212  positive_count++;
213  else
214  negative_count++;
215  }
216 
217  // Chi-square computation, compare against critical value (p=0.01)
218  double expected_count = static_cast<double>(samples) / 2.0;
219  double chi_square =
220  pow(positive_count - expected_count, 2.0) / expected_count +
221  pow(negative_count - expected_count, 2.0) / expected_count;
222  SOPHUS_TEST(passed, chi_square < 6.635);
223  }
224 
225  return passed;
226  }
