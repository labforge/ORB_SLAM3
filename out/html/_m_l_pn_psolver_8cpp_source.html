102  bNoMore = false;
103  vbInliers.clear();
104  nInliers=0;
105 
106  if(N<mRansacMinInliers)
107  {
108  bNoMore = true;

110  }

113 
114  int nCurrentIterations = 0;
115  while(mnIterations<mRansacMaxIts || nCurrentIterations<nIterations)
116  {
117  nCurrentIterations++;
118  mnIterations++;
119 
120  vAvailableIndices = mvAllIndices;
121 
122  //Bearing vectors and 3D points used for this ransac iteration
123  bearingVectors_t bearingVecs(mRansacMinSet);

225  void MLPnPsolver::SetRansacParameters(double probability, int minInliers, int maxIterations, int minSet, float epsilon, float th2){
226  mRansacProb = probability;
227  mRansacMinInliers = minInliers;
228  mRansacMaxIts = maxIterations;
229  mRansacEpsilon = epsilon;
230  mRansacMinSet = minSet;
231 
232  N = mvP2D.size(); // number of correspondences
233 
234  mvbInliersi.resize(N);
235 
236  // Adjust Parameters according to number of correspondences
237  int nMinInliers = N*mRansacEpsilon;
238  if(nMinInliers<mRansacMinInliers)
239  nMinInliers=mRansacMinInliers;
240  if(nMinInliers<minSet)
241  nMinInliers=minSet;
242  mRansacMinInliers = nMinInliers;
243 
244  if(mRansacEpsilon<(float)mRansacMinInliers/N)
245  mRansacEpsilon=(float)mRansacMinInliers/N;
246 
247  // Set RANSAC iterations according to probability, epsilon, and max iterations
248  int nIterations;
249 
250  if(mRansacMinInliers==N)
251  nIterations=1;
252  else
253  nIterations = ceil(log(1-mRansacProb)/log(1-pow(mRansacEpsilon,3)));
254 
255  mRansacMaxIts = max(1,min(nIterations,mRansacMaxIts));
256 
257  mvMaxError.resize(mvSigma2.size());
258  for(size_t i=0; i<mvSigma2.size(); i++)
259  mvMaxError[i] = mvSigma2[i]*th2;
260  }

356  void MLPnPsolver::computePose(const bearingVectors_t &f, const points_t &p, const cov3_mats_t &covMats,
357  const std::vector<int> &indices, transformation_t &result) {
358  size_t numberCorrespondences = indices.size();
359  assert(numberCorrespondences > 5);
360 
361  bool planar = false;
362  // compute the nullspace of all vectors
363  std::vector<Eigen::MatrixXd> nullspaces(numberCorrespondences);
364  Eigen::MatrixXd points3(3, numberCorrespondences);
365  points_t points3v(numberCorrespondences);
366  points4_t points4v(numberCorrespondences);
367  for (size_t i = 0; i < numberCorrespondences; i++) {
368  bearingVector_t f_current = f[indices[i]];
369  points3.col(i) = p[indices[i]];
370  // nullspace of right vector
371  Eigen::JacobiSVD<Eigen::MatrixXd, Eigen::HouseholderQRPreconditioner>
372  svd_f(f_current.transpose(), Eigen::ComputeFullV);
373  nullspaces[i] = svd_f.matrixV().block(0, 1, 3, 2);
374  points3v[i] = p[indices[i]];
375  }
376 
378  // 1. test if we have a planar scene
380 
381  Eigen::Matrix3d planarTest = points3 * points3.transpose();
382  Eigen::FullPivHouseholderQR<Eigen::Matrix3d> rankTest(planarTest);
383  Eigen::Matrix3d eigenRot;
384  eigenRot.setIdentity();
385 
386  // if yes -> transform points to new eigen frame
387  //if (minEigenVal < 1e-3 || minEigenVal == 0.0)
388  //rankTest.setThreshold(1e-10);
389  if (rankTest.rank() == 2) {
390  planar = true;
391  // self adjoint is faster and more accurate than general eigen solvers
392  // also has closed form solution for 3x3 self-adjoint matrices
393  // in addition this solver sorts the eigenvalues in increasing order
394  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> eigen_solver(planarTest);
395  eigenRot = eigen_solver.eigenvectors().real();
396  eigenRot.transposeInPlace();
397  for (size_t i = 0; i < numberCorrespondences; i++)
398  points3.col(i) = eigenRot * points3.col(i);
399  }
401  // 2. stochastic model
403  Eigen::SparseMatrix<double> P(2 * numberCorrespondences,
404  2 * numberCorrespondences);
405  bool use_cov = false;
406  P.setIdentity(); // standard
407 
408  // if we do have covariance information
409  // -> fill covariance matrix
410  if (covMats.size() == numberCorrespondences) {
411  use_cov = true;
412  int l = 0;
413  for (size_t i = 0; i < numberCorrespondences; ++i) {
414  // invert matrix
415  cov2_mat_t temp = nullspaces[i].transpose() * covMats[i] * nullspaces[i];
416  temp = temp.inverse().eval();
417  P.coeffRef(l, l) = temp(0, 0);
418  P.coeffRef(l, l + 1) = temp(0, 1);
419  P.coeffRef(l + 1, l) = temp(1, 0);
420  P.coeffRef(l + 1, l + 1) = temp(1, 1);
421  l += 2;
422  }
423  }
424 
426  // 3. fill the design matrix A
428  const int rowsA = 2 * numberCorrespondences;
429  int colsA = 12;
430  Eigen::MatrixXd A;
431  if (planar) {
432  colsA = 9;
433  A = Eigen::MatrixXd(rowsA, 9);
434  } else
435  A = Eigen::MatrixXd(rowsA, 12);
436  A.setZero();
437 
438  // fill design matrix
439  if (planar) {
440  for (size_t i = 0; i < numberCorrespondences; ++i) {
441  point_t pt3_current = points3.col(i);
442 
443  // r12
444  A(2 * i, 0) = nullspaces[i](0, 0) * pt3_current[1];
445  A(2 * i + 1, 0) = nullspaces[i](0, 1) * pt3_current[1];
446  // r13
447  A(2 * i, 1) = nullspaces[i](0, 0) * pt3_current[2];
448  A(2 * i + 1, 1) = nullspaces[i](0, 1) * pt3_current[2];
449  // r22
450  A(2 * i, 2) = nullspaces[i](1, 0) * pt3_current[1];
451  A(2 * i + 1, 2) = nullspaces[i](1, 1) * pt3_current[1];
452  // r23
453  A(2 * i, 3) = nullspaces[i](1, 0) * pt3_current[2];
454  A(2 * i + 1, 3) = nullspaces[i](1, 1) * pt3_current[2];
455  // r32
456  A(2 * i, 4) = nullspaces[i](2, 0) * pt3_current[1];
457  A(2 * i + 1, 4) = nullspaces[i](2, 1) * pt3_current[1];
458  // r33
459  A(2 * i, 5) = nullspaces[i](2, 0) * pt3_current[2];
460  A(2 * i + 1, 5) = nullspaces[i](2, 1) * pt3_current[2];
461  // t1
462  A(2 * i, 6) = nullspaces[i](0, 0);
463  A(2 * i + 1, 6) = nullspaces[i](0, 1);
464  // t2
465  A(2 * i, 7) = nullspaces[i](1, 0);
466  A(2 * i + 1, 7) = nullspaces[i](1, 1);
467  // t3
468  A(2 * i, 8) = nullspaces[i](2, 0);
469  A(2 * i + 1, 8) = nullspaces[i](2, 1);
470  }
471  } else {
472  for (size_t i = 0; i < numberCorrespondences; ++i) {
473  point_t pt3_current = points3.col(i);
474 
475  // r11
476  A(2 * i, 0) = nullspaces[i](0, 0) * pt3_current[0];
477  A(2 * i + 1, 0) = nullspaces[i](0, 1) * pt3_current[0];
478  // r12
479  A(2 * i, 1) = nullspaces[i](0, 0) * pt3_current[1];
480  A(2 * i + 1, 1) = nullspaces[i](0, 1) * pt3_current[1];
481  // r13
482  A(2 * i, 2) = nullspaces[i](0, 0) * pt3_current[2];
483  A(2 * i + 1, 2) = nullspaces[i](0, 1) * pt3_current[2];
484  // r21
485  A(2 * i, 3) = nullspaces[i](1, 0) * pt3_current[0];
486  A(2 * i + 1, 3) = nullspaces[i](1, 1) * pt3_current[0];
487  // r22
488  A(2 * i, 4) = nullspaces[i](1, 0) * pt3_current[1];
489  A(2 * i + 1, 4) = nullspaces[i](1, 1) * pt3_current[1];
490  // r23
491  A(2 * i, 5) = nullspaces[i](1, 0) * pt3_current[2];
492  A(2 * i + 1, 5) = nullspaces[i](1, 1) * pt3_current[2];
493  // r31
494  A(2 * i, 6) = nullspaces[i](2, 0) * pt3_current[0];
495  A(2 * i + 1, 6) = nullspaces[i](2, 1) * pt3_current[0];
496  // r32
497  A(2 * i, 7) = nullspaces[i](2, 0) * pt3_current[1];
498  A(2 * i + 1, 7) = nullspaces[i](2, 1) * pt3_current[1];
499  // r33
500  A(2 * i, 8) = nullspaces[i](2, 0) * pt3_current[2];
501  A(2 * i + 1, 8) = nullspaces[i](2, 1) * pt3_current[2];
502  // t1
503  A(2 * i, 9) = nullspaces[i](0, 0);
504  A(2 * i + 1, 9) = nullspaces[i](0, 1);
505  // t2
506  A(2 * i, 10) = nullspaces[i](1, 0);
507  A(2 * i + 1, 10) = nullspaces[i](1, 1);
508  // t3
509  A(2 * i, 11) = nullspaces[i](2, 0);
510  A(2 * i + 1, 11) = nullspaces[i](2, 1);
511  }
512  }
513 
515  // 4. solve least squares
517  Eigen::MatrixXd AtPA;
518  if (use_cov)
519  AtPA = A.transpose() * P * A; // setting up the full normal equations seems to be unstable
520  else
521  AtPA = A.transpose() * A;
522 
523  Eigen::JacobiSVD<Eigen::MatrixXd> svd_A(AtPA, Eigen::ComputeFullV);
524  Eigen::MatrixXd result1 = svd_A.matrixV().col(colsA - 1);
525 
527  // now we treat the results differently,
528  // depending on the scene (planar or not)
530  rotation_t Rout;
531  translation_t tout;
532  if (planar) // planar case
533  {
534  rotation_t tmp;
535  // until now, we only estimated
536  // row one and two of the transposed rotation matrix
537  tmp << 0.0, result1(0, 0), result1(1, 0),
538  0.0, result1(2, 0), result1(3, 0),
539  0.0, result1(4, 0), result1(5, 0);
540  // row 3
541  tmp.col(0) = tmp.col(1).cross(tmp.col(2));
542  tmp.transposeInPlace();
543 
544  double scale = 1.0 / std::sqrt(std::abs(tmp.col(1).norm() * tmp.col(2).norm()));
545  // find best rotation matrix in frobenius sense
546  Eigen::JacobiSVD<Eigen::MatrixXd> svd_R_frob(tmp, Eigen::ComputeFullU | Eigen::ComputeFullV);
547  rotation_t Rout1 = svd_R_frob.matrixU() * svd_R_frob.matrixV().transpose();
548  // test if we found a good rotation matrix
549  if (Rout1.determinant() < 0)
550  Rout1 *= -1.0;
551  // rotate this matrix back using the eigen frame
552  Rout1 = eigenRot.transpose() * Rout1;
553 
554  translation_t t = scale * translation_t(result1(6, 0), result1(7, 0), result1(8, 0));
555  Rout1.transposeInPlace();
556  Rout1 *= -1;
557  if (Rout1.determinant() < 0.0)
558  Rout1.col(2) *= -1;
559  // now we have to find the best out of 4 combinations
560  rotation_t R1, R2;
561  R1.col(0) = Rout1.col(0);
562  R1.col(1) = Rout1.col(1);
563  R1.col(2) = Rout1.col(2);
564  R2.col(0) = -Rout1.col(0);
565  R2.col(1) = -Rout1.col(1);
566  R2.col(2) = Rout1.col(2);
567 
568  vector<transformation_t, Eigen::aligned_allocator<transformation_t>> Ts(4);
569  Ts[0].block<3, 3>(0, 0) = R1;
570  Ts[0].block<3, 1>(0, 3) = t;
571  Ts[1].block<3, 3>(0, 0) = R1;
572  Ts[1].block<3, 1>(0, 3) = -t;
573  Ts[2].block<3, 3>(0, 0) = R2;
574  Ts[2].block<3, 1>(0, 3) = t;
575  Ts[3].block<3, 3>(0, 0) = R2;
576  Ts[3].block<3, 1>(0, 3) = -t;
577 
578  vector<double> normVal(4);
579  for (int i = 0; i < 4; ++i) {
580  point_t reproPt;
581  double norms = 0.0;
582  for (int p = 0; p < 6; ++p) {

594  } else // non-planar

597  tmp << result1(0, 0), result1(3, 0), result1(6, 0),
598  result1(1, 0), result1(4, 0), result1(7, 0),

608  if (Rout.determinant() < 0)
609  Rout *= -1.0;
610  // scale translation
611  tout = Rout * (scale * translation_t(result1(9, 0), result1(10, 0), result1(11, 0)));

636 
637  }

649  minx[5] = tout[2];

654  tout = translation_t(minx[3], minx[4], minx[5]);

656  result.block<3, 3>(0, 0) = Rout;
657  result.block<3, 1>(0, 3) = tout;

660  Eigen::Matrix3d MLPnPsolver::rodrigues2rot(const Eigen::Vector3d &omega) {
661  rotation_t R = Eigen::Matrix3d::Identity();
662 
663  Eigen::Matrix3d skewW;
664  skewW << 0.0, -omega(2), omega(1),
665  omega(2), 0.0, -omega(0),
666  -omega(1), omega(0), 0.0;
667 
668  double omega_norm = omega.norm();
669 
670  if (omega_norm > std::numeric_limits<double>::epsilon())
671  R = R + sin(omega_norm) / omega_norm * skewW
672  + (1 - cos(omega_norm)) / (omega_norm * omega_norm) * (skewW * skewW);
673 
674  return R;
675  }

677  Eigen::Vector3d MLPnPsolver::rot2rodrigues(const Eigen::Matrix3d &R) {
678  rodrigues_t omega;
679  omega << 0.0, 0.0, 0.0;
680 
681  double trace = R.trace() - 1.0;
682  double wnorm = acos(trace / 2.0);
683  if (wnorm > std::numeric_limits<double>::epsilon())
684  {
685  omega[0] = (R(2, 1) - R(1, 2));
686  omega[1] = (R(0, 2) - R(2, 0));
687  omega[2] = (R(1, 0) - R(0, 1));
688  double sc = wnorm / (2.0*sin(wnorm));
689  omega *= sc;
690  }
691  return omega;
692  }

694  void MLPnPsolver::mlpnp_gn(Eigen::VectorXd &x, const points_t &pts, const std::vector<Eigen::MatrixXd> &nullspaces,
695  const Eigen::SparseMatrix<double> Kll, bool use_cov) {
696  const int numObservations = pts.size();
697  const int numUnknowns = 6;
698  // check redundancy
699  assert((2 * numObservations - numUnknowns) > 0);
700 
701  // =============
702  // set all matrices up
703  // =============
704 
705  Eigen::VectorXd r(2 * numObservations);
706  Eigen::VectorXd rd(2 * numObservations);
707  Eigen::MatrixXd Jac(2 * numObservations, numUnknowns);
708  Eigen::VectorXd g(numUnknowns, 1);
709  Eigen::VectorXd dx(numUnknowns, 1); // result vector
710 
711  Jac.setZero();
712  r.setZero();
713  dx.setZero();
714  g.setZero();
715 
716  int it_cnt = 0;
717  bool stop = false;
718  const int maxIt = 5;
719  double epsP = 1e-5;
720 
721  Eigen::MatrixXd JacTSKll;
722  Eigen::MatrixXd A;
723  // solve simple gradient descent
724  while (it_cnt < maxIt && !stop) {
725  mlpnp_residuals_and_jacs(x, pts,
726  nullspaces,
727  r, Jac, true);
728 
729  if (use_cov)
730  JacTSKll = Jac.transpose() * Kll;
731  else
732  JacTSKll = Jac.transpose();
733 
734  A = JacTSKll * Jac;
735 
736  // get system matrix
737  g = JacTSKll * r;
738 
739  // solve
740  Eigen::LDLT<Eigen::MatrixXd> chol(A);
741  dx = chol.solve(g);
742  // this is to prevent the solution from falling into a wrong minimum
743  // if the linear estimate is spurious
744  if (dx.array().abs().maxCoeff() > 5.0 || dx.array().abs().minCoeff() > 1.0)
745  break;
746  // observation update
747  Eigen::MatrixXd dl = Jac * dx;
748  if (dl.array().abs().maxCoeff() < epsP) {
749  stop = true;
750  x = x - dx;
751  break;
752  } else
753  x = x - dx;
754 
755  ++it_cnt;
756  }//while
757  // result
758  }
